// Lexicographically smallest string of length N and sum K

class Solution {  // greedy Time and space = O(N)
    public String getSmallestString(int n, int k) {
        char[] res = new char[n];
        Arrays.fill(res, 'a');

        int idx = n-1;
        k -= n;          // since all 'a's contribute 1 point each

        while(k > 0) {
            int add = Math.min(25, k);
            res[idx] = (char) ('a' + add);
            idx--;
            k -= add;
        }

        return new String(res);
    }
}
