class MinStack {   //O(1);
    Stack<Integer> main;
    Stack<Integer> min;

    public MinStack() {
        main=new Stack<>();
        min=new Stack<>();
    }
    
    public void push(int val) {
        main.push(val);
        if(min.isEmpty()){
            min.push(val);
        }else{
            min.push(Math.min(val,min.peek()));
        }
    }
    
    public void pop() {
        if (!main.isEmpty()) {
            main.pop();
            min.pop();
        }
    }
    
    public int top() {
       return main.peek();
    }
    
    public int getMin() {
       return min.peek();
    }
}
