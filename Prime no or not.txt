import java.lang.Math;

public class pattern {
    public static boolean primeNo(int n) {
        boolean isPrime = true;
        for(int i=2; i<= Math.sqrt(n); i++) {
            if(n % i == 0) {
                isPrime = false;
                break;
            }
        }
        return isPrime;
    }

    public static void main(String[] args) {
        int n = 53;

        for(int i=2; i<n; i++) {
            if(primeNo(i)) {
                System.out.print(i + " ");
            }
        }
           
    }
}

Time complexity = O(root n)
